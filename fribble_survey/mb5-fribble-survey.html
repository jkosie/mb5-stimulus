base_averages <- base_data %>% 
  group_by(item) %>% 
  summarise(avg_rating = mean(rating, na.rm = TRUE))

base_data <- base_data %>% 
  left_join(base_averages) #to allow for sorting individual data points
## Joining with `by = join_by(item)`
base_averages %>% 
  ggplot(aes(x = item, y = avg_rating, fill = item)) +
  geom_bar(stat = "identity") +
  geom_point(data = base_data, aes(x = reorder(item, -avg_rating), y = rating), position=position_jitter(width =.15), size = 2, shape = 1) +
  theme(legend.position = "none")
